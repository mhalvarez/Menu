CAMBIAR LOS APUNTES DE "TOTAL LIQUIDO"   POR LOS APUNTES  DE ESTE BUCLE

OJO ESTA SRPIT ES PARA EL ASIENTO 3   HACER LO MISMOPOR CADA TIPO DE "VENTA LIQUIDA"


SELECT  NVL (SERV_CTB1,'0') CUENTA,SERV_DESC DEPARTAMENTO,ROUND(SUM (MOVI_VCRE-MOVI_IMP1),2) TOTAL
    FROM TNHT_MOVI, TNHT_FACT, TNHT_FAMO,TNHT_SERV
   WHERE TNHT_FAMO.FACT_CODI = TNHT_FACT.FACT_CODI
     AND TNHT_FAMO.SEFA_CODI = TNHT_FACT.SEFA_CODI
     AND TNHT_FAMO.MOVI_DARE = TNHT_MOVI.MOVI_DARE
     AND TNHT_FAMO.MOVI_CODI = TNHT_MOVI.MOVI_CODI
     AND TNHT_FACT.FACT_STAT = '1'
     AND FACT_CLEN = '1'
   --  AND TNHT_MOVI.MOVI_DATR = '30/09/2007'
   --  AND TNHT_MOVI.MOVI_DAVA = '30/09/2007'
     AND TNHT_FACT.FACT_DAEM = '30/09/2007'
     AND (FACT_DAAN IS NULL OR FACT_DAEM < FACT_DAAN)
	 AND TNHT_MOVI.SERV_CODI = TNHT_SERV.SERV_CODI
GROUP BY SERV_CTB1,SERV_DESC
