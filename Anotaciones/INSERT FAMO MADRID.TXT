INSERT INTO TNHT_FAMO (FACT_CODI,SEFA_CODI,MOVI_CODI,MOVI_DARE) 
SELECT M.FACT_CODI,M.SEFA_CODI,M.MOVI_CODI,M.MOVI_DARE 
FROM TNHT_MOVI M, 
(SELECT FACT_CODI,SEFA_CODI FROM ( 
  SELECT TNHT_FACT.FACT_CODI,TNHT_FACT.SEFA_CODI,COUNT(TNHT_FAMO.FACT_CODI) CUANTOS 
    FROM TNHT_FACT,TNHT_FAMO 
   WHERE TNHT_FACT.FACT_CODI=TNHT_FAMO.FACT_CODI(+) 
     AND TNHT_FACT.SEFA_CODI=TNHT_FAMO.SEFA_CODI(+) 
  GROUP BY TNHT_FACT.FACT_CODI,TNHT_FACT.SEFA_CODI) 
 WHERE CUANTOS = 0) F 
WHERE M.FACT_CODI=F.FACT_CODI 
  AND M.SEFA_CODI=F.SEFA_CODI
