SELECT TNHT_MOVI.MOVI_VLIQ,TNPL_MOVI.MOVI_VLIQ,
  TNHT_MOVI.SERV_CODI,
  TNHT_MOVI.MOVI_CODI,
  TNPL_MOVI.NEWH_CODI,
  TNHT_MOVI.MOVI_DARE,
  TNPL_MOVI.NEWH_DARE,
  TNHT_SERV.SERV_DESC,TNHT_MOVI.MOVI_DESC,ADIC_DESC,TNPL_MOVI.ADIC_CODI
FROM RMC.TNHT_MOVI,
GMS.TNPL_ADIC,
  GMS.TNPL_MOVI,
  RMC.TNHT_SERV
WHERE TNHT_MOVI.SERV_CODI = TNHT_SERV.SERV_CODI
AND TNHT_MOVI.MOVI_CODI = TNPL_MOVI.NEWH_CODI
 AND TNHT_MOVI.MOVI_DARE = TNPL_MOVI.NEWH_DARE
 AND TNPL_MOVI.ADIC_CODI = TNPL_ADIC.ADIC_CODI
 AND TNPL_MOVI.ADIC_CODI IN
  (SELECT ADIC_CODI
   FROM GMS.TNPL_ADIC
   WHERE ADIC_TIPO = 3)
